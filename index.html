using UnityEngine; using UnityEngine.EventSystems; public class ShapeDragDrop : MonoBehaviour, IBeginDragHandler, IDragHandler, IEndDragHandler { private Vector3 startPos; private CanvasGroup canvasGroup; // Gán hình đúng lỗ này trong Inspector public Transform targetSlot; public AudioClip correctSound; public AudioClip wrongSound; private AudioSource audioSource; void Awake() { canvasGroup = gameObject.AddComponent(); startPos = transform.position; audioSource = gameObject.AddComponent(); } public void OnBeginDrag(PointerEventData eventData) { startPos = transform.position; canvasGroup.blocksRaycasts = false; } public void OnDrag(PointerEventData eventData) { transform.position = eventData.position; } public void OnEndDrag(PointerEventData eventData) { canvasGroup.blocksRaycasts = true; // Kiểm tra khoảng cách với lỗ float distance = Vector3.Distance(transform.position, targetSlot.position); if(distance < 50f) // khoảng cách hợp lệ để đặt đúng { transform.position = targetSlot.position; audioSource.PlayOneShot(correctSound); Debug.Log("Đúng hình!"); } else { transform.position = startPos; audioSource.PlayOneShot(wrongSound); Debug.Log("Sai hình, thử lại!"); } } }
